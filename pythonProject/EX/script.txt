def maior_primo(n):
    i = 0
    cont = 0
    if(n >= 2):
        while i <= n or cont < 2:
            i = i + 1
            x = n % i
            if x == 0:
                cont = cont + 1

        if cont <= 2:
            print("primo")
        else:
            print("não primo")

maior_primo(100)
